To download the dataset
-----------------------
Local Machine:
scp get_download_links.py arung@130.127.133.141:/mydata/Kinetics
scp download.py arung@130.127.133.141:/mydata/Kinetics

Server:
# Setup youtube-dl and ffmpeg
sudo curl -L https://yt-dl.org/downloads/latest/youtube-dl -o /usr/local/bin/youtube-dl
sudo chmod a+rx /usr/local/bin/youtube-dl
sudo apt-get install -y ffmpeg

mkdir /mydata/Kinetics
cd /mydata/Kinetics && wget https://storage.googleapis.com/deepmind-media/Datasets/kinetics400.tar.gz && tar -zxvf kinetics400.tar.gz
cd /mydata/Kinetics && python3 get_download_links.py
cd /mydata/Kinetics && mkdir Kinetics400_Splits && cp Kinetics_Train.csv Kinetics400_Splits/
cd /mydata/Kinetics/Kinetics400_Splits && split -l 10000 Kinetics_Train.csv

# Change the path of the file to download
python3 download.py
